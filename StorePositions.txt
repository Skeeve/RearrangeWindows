use scripting additions
use framework "Foundation"

on saveFolderName()
	return "StorePositions"
end saveFolderName

on run
	-- Selects whether the user wants to save or restore the positions
	set action to button returned of (display dialog "What would you like to do?" buttons {"Cancel", "Restore", "Save"} default button "Restore" with title (name of me))

	if action is "Save" then
		saveWindowPositions()
	else if action is "Restore" then
		restoreWindowPositions()
	end if
end run

-- Function for saving the window positions
on saveWindowPositions()
	set NSJSONWritingPrettyPrinted to a reference to 1
	set NSJSONSerialization to a reference to current application's NSJSONSerialization
	set processNames to createSaveFolderAndConfig()
	set positionList to {}

	-- Asks for a user-defined name, e.g. "Home" or "Work"
	set userName to text returned of (display dialog "Enter a name for this position (e.g., 'Home', 'Work'):" default answer "Home" with title (name of me))

	-- Run a loop over the windows from the applications
	repeat with proc in processNames
		--try
		tell application "System Events" to tell process proc
			set its frontmost to true
			tell window 1
				set pos to get its position
				set sze to get its size
				set newPos to {|name|:proc, |position|:pos, |size|:sze}
				set the end of positionList to newPos
			end tell
		end tell
		--end try
	end repeat

	-- Determines the folder for the saved files
	set saveFolder to (path to application support from user domain as text) & saveFolderName() & ":"

	-- If the folder does not exist, it is created
	do shell script "mkdir -p " & quoted form of POSIX path of saveFolder

	-- Selects the storage location and saves the file
	set savePath to saveFolder & userName & ".json"

	-- Convert positionList to JSON data
	set theJSONData to NSJSONSerialization's dataWithJSONObject:positionList options:NSJSONWritingPrettyPrinted |error|:(missing value)

	-- Write the JSON data to the file
	theJSONData's writeToFile:(savePath's POSIX path) atomically:true

	-- Bring to the foreground now
	tell application "System Events"
		set frontmost of process (name of me) to true
	end tell

	display dialog "Window positions for '" & userName & "' successfully saved!" buttons {"OK"} default button "OK" with title (name of me)
end saveWindowPositions

-- Function for restoring the window positions
on restoreWindowPositions()
	-- Determines the folder where the JSON files are stored
	set saveFolder to (path to application support from user domain as text) & saveFolderName() & ":"

	-- Check if the folder exists
	try
		do shell script "test -d " & quoted form of POSIX path of saveFolder

	on error
		display dialog "There is no 'SavePositions' folder. No saved window positions found." buttons {"OK"} default button "OK" with title (name of me)
		return
	end try

	-- Select a JSON file from the SavePositions folder
	set fileToRestore to choose file with prompt "Select the file with the saved window positions:" default location (saveFolder as alias) of type {"public.json"} -- Only allow JSON files

	-- Read JSON data from the selected file
	try
		set fileRef to open for access fileToRestore
		set rawData to read fileRef
		close access fileRef
	on error
		display alert (name of me) message "Error: Could not open file." buttons {"OK"} default button "OK"
		return
	end try

	-- Convert JSON data to AppleScript objects
	set theJSON to readJSON(rawData)

	-- Restore the window positions
	repeat with itm in theJSON

		set procName to (|name| of itm) as text
		set posX to (item 1 of (position of itm)) as integer
		set posY to (item 2 of (position of itm)) as integer
		set sizeW to (item 1 of (|size| of itm)) as integer
		set sizeH to (item 2 of (|size| of itm)) as integer

		-- Now restore the position and size of the window
		tell application "System Events"
			tell process procName
				set its frontmost to true
				-- Check if the window exists
				if exists window 1 then
					-- Ensure position and size are treated as integers
					try
						set position of window 1 to {posX, posY}
						set size of window 1 to {sizeW, sizeH}
					on error errMsg
						display alert (name of me) message "Error restoring window position for " & procName & ": " & errMsg buttons {"OK"} default button "OK"
					end try
				else
					display alert (name of me) message "Window of " & procName & " not found." buttons {"OK"} default button "OK"
				end if
			end tell
		end tell
	end repeat

	-- display dialog "Window positions successfully restored!" buttons {"OK"} default button "OK"
end restoreWindowPositions

-- Function for reading JSON data
on readJSON(strJSON)
	set ca to current application
	set {x, e} to ca's NSJSONSerialization's JSONObjectWithData:((ca's NSString's stringWithString:strJSON)'s dataUsingEncoding:(ca's NSUTF8StringEncoding)) options:0 |error|:(reference)

	-- Error handling if the JSON is not correct
	if x is missing value then
		error e's localizedDescription() as text
	else
		return x -- returns the deserialized object
	end if
end readJSON

-- Handler to create the folder and configuration file
on createSaveFolderAndConfig()
	set saveFolder to (path to application support from user domain as text) & saveFolderName() & ":"

	set qFolder to quoted form of POSIX path of saveFolder
	set qConf to quoted form of (saveFolderName() & ".conf")
	set commands to "mkdir -p " & qFolder & " && cd " & qFolder & " && [[ ! -r " & qConf & " ]] && cat <<CONF>> " & qConf & linefeed & "Calendar" & return & linefeed & "Mail" & return & linefeed & "CONF" & linefeed & "cat " & qConf
	return paragraphs of (do shell script commands)
end createSaveFolderAndConfig
